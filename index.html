  <title>声音花园</title>

  </style>

<body>

  <script src="https://cdn.bootcdn.net/ajax/libs/p5.js/1.4.0/p5.min.js"></script>

  <script src="https://cdn.bootcdn.net/ajax/libs/p5.js/1.4.0/addons/p5.sound.min.js"></script>

  <script>

    // 全局变量

    let mic, fft;

    let raindrops = [];

    let ripples = [];

    let flowers = [];

    let threshold = 0.1;

    let startMsg = "轻点屏幕开始";

 

    // 初始化

    function setup() {

      let canvas = createCanvas(windowWidth, windowHeight);

      canvas.mousePressed(startAudio);

      canvas.touchStarted(startAudio); // 手机触摸支持

      colorMode(HSB, 360, 100, 100, 1);

      textAlign(CENTER, CENTER);

      try {

        mic = new p5.AudioIn();

        mic.start();

        fft = new p5.FFT();

        fft.setInput(mic);

      } catch (e) {

        startMsg = "请允许麦克风权限";

      }

    }

 

    // 音频激活

    function startAudio() {

      userStartAudio();

      startMsg = null;

    }

 

    // 主绘制循环

    function draw() {

      background(10, 20, 15);

      if (startMsg) {

        fill(255);

        textSize(20);

        text(startMsg, width / 2, height / 2);

        return;

      }

      // 获取音量

      let volume = mic.getLevel();

      // 声音触发雨滴

      if (volume > threshold) {

        raindrops.push(new RainDrop(random(width), 0, random(5, 10)));

      }

      // 更新雨滴

      for (let i = raindrops.length - 1; i >= 0; i--) {

        raindrops[i].update();

        raindrops[i].show();

        if (raindrops[i].hitsGround()) {

          ripples.push(new Ripple(raindrops[i].x, height - 100));

          raindrops.splice(i, 1);

        } else if (raindrops[i].offScreen()) {

          raindrops.splice(i, 1);

        }

      }

      // 更新涟漪

      for (let i = ripples.length - 1; i >= 0; i--) {

        ripples[i].update();

        ripples[i].show();

        if (ripples[i].isFinished()) {

          flowers.push(new Flower(ripples[i].x, height - 100, random(100, 300)));

          ripples.splice(i, 1);

        }

      }

      // 更新花朵

      for (let flower of flowers) {

        flower.grow();

        flower.displayStem();

        flower.displayFlower();

      }

    }

 

    // 雨滴类

    class RainDrop {

      constructor(x, y, size) {

        this.x = x;

        this.y = y;

        this.size = size;

        this.speed = random(6, 9);

      }

      update() { this.y += this.speed; }

      show() {

        noStroke();

        fill(200, 80, 100);

        beginShape();

        vertex(this.x, this.y);

        vertex(this.x + this.size / 2, this.y + this.size);

        vertex(this.x, this.y + this.size * 1.5);

        vertex(this.x - this.size / 2, this.y + this.size);

        endShape(CLOSE);

        ellipse(this.x, this.y + this.size, this.size, this.size * 0.9);

      }

      hitsGround() { return this.y >= height - 100; }

      offScreen() { return this.y > height; }

    }

 

    // 涟漪类

    class Ripple {

      constructor(x, y) {

        this.x = x;

        this.y = y;

        this.radius = 0;

        this.alpha = 200;

      }

      update() { this.radius += 2; this.alpha -= 4; }

      show() {

        noFill();

        stroke(200, 200, 255, this.alpha);

        strokeWeight(2);

        ellipse(this.x, this.y, this.radius * 2, this.radius / 2);

      }

      isFinished() { return this.alpha <= 0; }

    }

 

    // 花朵类

    class Flower {

      constructor(x, y, stemHeight) {

        this.pos = createVector(x, y);

        this.stemHeight = stemHeight;

        this.stemGrowth = 0;

        this.stemRate = random(2, 5);

        this.bloomed = false;

        this.petalSize = 0;

        this.maxPetalSize = random(15, 25);

        this.petalColor = random() < 0.3 ? color(0, 0, 100) : color(random(360), 80, 100);

        this.centerColor = color(60, 80, 100);

        this.petalCount = int(random(20, 30));

      }

      grow() {

        if (!this.bloomed) {

          this.stemGrowth += this.stemRate;

          if (this.stemGrowth >= this.stemHeight) this.bloomed = true;

        } else if (this.petalSize < this.maxPetalSize) {

          this.petalSize += 0.5;

        }

      }

      displayStem() {

        push();

        translate(this.pos.x, this.pos.y);

        stroke(15, 80, 60);

        strokeWeight(2);

        line(0, 0, 0, -this.stemGrowth);

        pop();

      }

      displayFlower() {

        if (!this.bloomed) return;

        push();

        translate(this.pos.x, this.pos.y - this.stemGrowth);

        rotate(frameCount * 0.01);

        noStroke();

        fill(this.petalColor);

        for (let i = 0; i < this.petalCount; i++) {

          const angle = map(i, 0, this.petalCount, 0, TWO_PI);

          push();

          rotate(angle);

          ellipse(0, this.petalSize / 2, this.petalSize / 4, this.petalSize * 1.5);

          pop();

        }

        fill(this.centerColor);

        ellipse(0, 0, this.maxPetalSize / 3, this.maxPetalSize / 3);

        pop();

      }

    }

 

    // 窗口大小变化

    function windowResized() {

      resizeCanvas(windowWidth, windowHeight);

    }

  </script>

</body>
